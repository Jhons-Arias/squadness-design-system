<script>
var icDragged = null;
var icSource  = null;

function icDragStart(e) {
  icDragged = e.currentTarget;
  icSource  = e.currentTarget.closest('.sq-items-container__body');
  e.currentTarget.classList.add('is-dragging');
  e.dataTransfer.effectAllowed = 'move';
}
function icDragEnd(e) {
  e.currentTarget.classList.remove('is-dragging');
}
function icDragOver(container) {
  container.classList.add('drag-over');
}
function icDragLeave(e, container) {
  if (!container.contains(e.relatedTarget)) {
    container.classList.remove('drag-over');
  }
}
function icDrop(e, container) {
  e.preventDefault();
  container.classList.remove('drag-over');
  if (!icDragged || container === icSource) return;

  var empty = container.querySelector('.sq-items-container__empty');
  if (empty) empty.remove();

  container.appendChild(icDragged);

  if (icSource && icSource.querySelectorAll('.sq-drag-item').length === 0) {
    var el = document.createElement('div');
    el.className = 'sq-items-container__empty';
    el.innerHTML = '<span class="sq-items-container__empty-text">Selecciona un equipo o ruta para comenzar</span>';
    icSource.appendChild(el);
  }
  icDragged = null; icSource = null;
}
</script>
